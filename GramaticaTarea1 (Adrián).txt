<navidad> ::= { <bolsanavideña> | <regalo> | <regaloprin> | <regalocomprado> | <regalomanual> }

<bolsanavideña> ::= "function" <identificador> "(" [<parametros>] ")" "{" <instrucciones> "}"

<parametros> ::= <tipo> <identificador> { "," <tipo> <identificador> }

<instrucciones> ::= { <instruccion> }

<instruccion> ::= <asignacion> | <lectura> | <escritura> | <bolsanavideña> | <declaracionlocal> | <comentario>

<asignacion> ::= <identificador> "=" <regalo> ";"
              | <identificador> "[" <regalo> "]" "=" <regalo> ";"

<lectura> ::= "read" "(" <identificador> ")"
          | "read" "(" <identificador> "[" <regalo> "]" ")"

<escritura> ::= "write" "(" <regalo> { "," <regalo> } ")" ";"
            | "write" "(" <identificador> "[" <regalo> "]" { "," <regalo> } ")" ";"

<declaracionlocal> ::= "local" <tipo> <identificador> [ "[" <entero> "]" ] [ "=" <inicializacion> ] ";"

<inicializacion> ::= <regalo> | "{" { <regalo> "," } <regalo> "}"

<comentario> ::= "@" <texto> | "/_" <texto> "_/"

<tipo> ::= "entero" | "char"

<regalo> ::= <regaloprin> | <regalocomprado> | <regalomanual>

<regaloprin> ::= <termino> { ("+" | "-") <termino> }

<termino> ::= <factor> { ("*" | "/") <factor> }

<factor> ::= <identificador> | <entero> | <char> | "(" <regalo> ")"

<regalocomprado> ::= <regaloprin> ( "==" | "!=" | "<" | ">" | "<=" | ">=" ) <regaloprin>

<regalomanual> ::= <regalocomprado> { "and" <regalocomprado> | "or" <regalocomprado> }

<identificador> ::= <letra> { <letra> | <digito> }

<letra> ::= "a" | "b" | ... | "z" | "A" | "B" | ... | "Z"

<digito> ::= "0" | "1" | ... | "9"

<entero> ::= <digito> { <digito> }

<char> ::= "'" <caracter> "'"

<caracter> ::= <letra> | <digito> | " " | "+" | "-" | "*" | "/" | "=" | "!" | "<" | ">" | "(" | ")" | "{" | "}" | "[" | "]" | "," | ";" | ":" | "." | "?" | "!"
